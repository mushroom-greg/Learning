Future<List<ImgurTag>> getTags ()

Gets tag list

Implementation 

Future<List<ImgurTag>> getTags() async {
  var res = await _http.createRequest("GET", "/tags");
  var tmp = new List<ImgurTag>();

  res.response['data']['tags'].forEach((o) {
    tmp.add(new ImgurTag(o));
  });

  return tmp;
}
